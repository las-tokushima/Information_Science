## A-4 スクリプトの記述

### R（p. 220）

# 標本分散と不偏分散を確認する
# xにデータを代入
x <- c(60, 65, 70, 80, 90)

# サンプルサイズをnに代入
n <- length(x)

# 算術平均値をmに代入
m <- mean(x)

# 偏差をdevに代入
dev <- x - m

# 標本分散
# 偏差の2乗の総和をサンプルサイズで除す
(sum(dev * dev) / n)

# 不偏分散
 # 偏差の2乗の総和を(サンプルサイズ - 1)で除す
(sum(dev * dev) / (n - 1))

# R組み込みのvar関数
var(x)
